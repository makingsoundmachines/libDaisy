159 void SwapBuffersAndTransmit()
160 {
161 // wait for current transmission to complete
162 while(current_driver_idx_ >= 0) {};
163
164 // swap buffers
165 auto tmp = transmit_buffer_;
166 transmit_buffer_ = draw_buffer_;
167 draw_buffer_ = tmp;
168
169 // copy current transmit buffer contents to the new draw buffer
170 // to keep the led settings (if required)
171 if(Is31fl3731_persistentBufferContents)
172 {
173 for(int d = 0; d < Is31fl3731_numDrivers; d++)
174 for(int ch = 0; ch < 145; ch++)
175 draw_buffer_[d].data[ch]
176 = transmit_buffer_[d].data[ch];
177 }
178
179 // start transmission
180 current_driver_idx_ = -1;
181 ContinueTransmission();
182 }
