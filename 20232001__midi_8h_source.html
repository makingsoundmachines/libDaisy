199 void Parse(uint8_t byte)
200 {
201 // reset parser when status byte is received
202 if((byte & kStatusByteMask) && pstate_ != ParserSysEx)
203 {
204 pstate_ = ParserEmpty;
205 }
206 switch(pstate_)
207 {
208 case ParserEmpty:
209 // check byte for valid Status Byte
210 if(byte & kStatusByteMask)
211 {
212 // Get MessageType, and Channel
213 incoming_message_.channel = byte & kChannelMask;
214 incoming_message_.type = static_cast<MidiMessageType>(
215 (byte & kMessageMask) >> 4);
216
217 // Validate, and move on.
218 if(incoming_message_.type < MessageLast)
219 {
220 pstate_ = ParserHasStatus;
221 // Mark this status byte as running_status
222 running_status_ = incoming_message_.type;
223
224 if(running_status_ == SystemCommon)
225 {
226 incoming_message_.channel = 0;
227 //system real time = 1111 1xxx
228 if(byte & 0x08)
229 {
230 incoming_message_.type = SystemRealTime;
231 running_status_ = SystemRealTime;
232 incoming_message_.srt_type
233 = static_cast<SystemRealTimeType>(
234 byte & kSystemRealTimeMask);
235
236 //short circuit to start
237 pstate_ = ParserEmpty;
238 // event_q_.Write(incoming_message_);
239 event_q_.PushBack(incoming_message_);
240 }
241 //system common
242 else
243 {
244 incoming_message_.sc_type
245 = static_cast<SystemCommonType>(byte
246 & 0x07);
247 //sysex
248 if(incoming_message_.sc_type == SystemExclusive)
249 {
250 pstate_ = ParserSysEx;
251 incoming_message_.sysex_message_len = 0;
252 }
253 //short circuit
254 else if(incoming_message_.sc_type > SongSelect)
255 {
256 pstate_ = ParserEmpty;
257 // event_q_.Write(incoming_message_);
258 event_q_.PushBack(incoming_message_);
259 }
260 }
261 }
262 }
263 // Else we'll keep waiting for a valid incoming status byte
264 }
265 else
266 {
267 // Handle as running status
268 incoming_message_.type = running_status_;
269 incoming_message_.data[0] = byte & kDataByteMask;
270 //check for single byte running status, really this only applies to channel pressure though
271 if(running_status_ == ChannelPressure
272 || running_status_ == ProgramChange
273 || incoming_message_.sc_type == MTCQuarterFrame
274 || incoming_message_.sc_type == SongSelect)
275 {
276 //Send the single byte update
277 pstate_ = ParserEmpty;
278 //event_q_.Write(incoming_message_);
279 event_q_.PushBack(incoming_message_);
280 }
281 else
282 {
283 pstate_
284 = ParserHasData0; //we need to get the 2nd byte yet.
285 }
286 }
287 break;
288 case ParserHasStatus:
289 if((byte & kStatusByteMask) == 0)
290 {
291 incoming_message_.data[0] = byte & kDataByteMask;
292 if(running_status_ == ChannelPressure
293 || running_status_ == ProgramChange
294 || incoming_message_.sc_type == MTCQuarterFrame
295 || incoming_message_.sc_type == SongSelect)
296 {
297 //these are just one data byte, so we short circuit back to start
298 pstate_ = ParserEmpty;
299 //event_q_.Write(incoming_message_);
300 event_q_.PushBack(incoming_message_);
301 }
302 else
303 {
304 pstate_ = ParserHasData0;
305 }
306
307 //ChannelModeMessages (reserved Control Changes)
308 if(running_status_ == ControlChange
309 && incoming_message_.data[0] > 119)
310 {
311 incoming_message_.type = ChannelMode;
312 running_status_ = ChannelMode;
313 incoming_message_.cm_type
314 = static_cast<ChannelModeType>(
315 incoming_message_.data[0] - 120);
316 }
317 }
318 else
319 {
320 // invalid message go back to start ;p
321 pstate_ = ParserEmpty;
322 }
323 break;
324 case ParserHasData0:
325 if((byte & kStatusByteMask) == 0)
326 {
327 incoming_message_.data[1] = byte & kDataByteMask;
328
329 //velocity 0 NoteOns are NoteOffs
330 if(running_status_ == NoteOn
331 && incoming_message_.data[1] == 0)
332 {
333 incoming_message_.type = NoteOff;
334 }
335
336 // At this point the message is valid, and we can add this MidiEvent to the queue
337 //event_q_.Write(incoming_message_);
338 event_q_.PushBack(incoming_message_);
339 }
340 else
341 {
342 // invalid message go back to start ;p
343 pstate_ = ParserEmpty;
344 }
345 // Regardless, of whether the data was valid or not we go back to empty
346 // because either the message is queued for handling or its not.
347 pstate_ = ParserEmpty;
348 break;
349 case ParserSysEx:
350 // end of sysex
351 if(byte == 0xf7)
352 {
353 pstate_ = ParserEmpty;
354 //event_q_.Write(incoming_message_);
355 event_q_.PushBack(incoming_message_);
356 }
357 else if(incoming_message_.sysex_message_len < SYSEX_BUFFER_LEN)
358 {
359 incoming_message_
360 .sysex_data[incoming_message_.sysex_message_len]
361 = byte;
362 incoming_message_.sysex_message_len++;
363 }
364 break;
365 default: break;
366 }
367 }
